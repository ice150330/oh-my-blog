**打字机效果实现（纯 CSS + JS）**：

```vue
<!-- components/home/HeroTypingText.vue -->
<script setup lang="ts">
// 打字机效果组件：循环打印传入的文本数组，支持打字/删除动画
const props = defineProps<{ texts: string[] }>();

const currentIndex = ref(0);      // 当前显示文本的索引
const displayedText = ref('');    // 当前已显示的文本片段
const isDeleting = ref(false);    // 是否处于删除状态

let timer: ReturnType<typeof setInterval>;

onMounted(() => {
  const text = props.texts[currentIndex.value];
  let charIndex = isDeleting.value ? text.length : 0;

  // 定时器：控制逐字打印与逐字删除的节奏
  timer = setInterval(() => {
    if (!isDeleting.value) {
      // 打字阶段：逐字追加
      displayedText.value = text.slice(0, charIndex + 1);
      charIndex++;
      if (charIndex === text.length) {
        // 打印完成后停顿 2 秒，进入删除阶段
        setTimeout(() => { isDeleting.value = true; }, 2000);
      }
    } else {
      // 删除阶段：逐字回退
      displayedText.value = text.slice(0, charIndex - 1);
      charIndex--;
      if (charIndex === 0) {
        // 删除完成后切换至下一条文本
        isDeleting.value = false;
        currentIndex.value = (currentIndex.value + 1) % props.texts.length;
      }
    }
  }, 80);
});

onUnmounted(() => clearInterval(timer));
</script>
```

<!-- components/ui/AnimatedCounter.vue -->
<script setup lang="ts">
import { gsap } from 'gsap';

// 数字滚动动画组件：从 0 滚动到目标值，适用于统计数据展示
const props = defineProps<{ target: number; suffix?: string; duration?: number }>();
const displayValue = ref(0);
const counterRef = ref<HTMLElement>();

onMounted(() => {
  // 使用 GSAP 驱动数字平滑递增
  gsap.to(displayValue, {
    value: props.target,
    duration: props.duration || 1.5,
    ease: 'power2.out',
    onUpdate: () => {
      displayValue.value = Math.round(displayValue.value);
    },
  });
});
</script>

### 5.2 分类切换动画

使用 FLIP 动画（First, Last, Invert, Play）——分类切换时卡片自动重排：

```typescript
// composables/useFlipAnimation.ts
import { gsap } from 'gsap';
import { Flip } from 'gsap/Flip';

export const useFlipAnimation = () => {
  const containerRef = ref<HTMLElement>();

  const animateFilter = () => {
    if (!containerRef.value) return;
    const state = Flip.getState('.project-card');
    // DOM 已更新（v-if 过滤后）
    Flip.from(state, {
      duration: 0.4,
      ease: 'power2.inOut',
      absolute: true,
    });
  };

  return { containerRef, animateFilter };
};
```
