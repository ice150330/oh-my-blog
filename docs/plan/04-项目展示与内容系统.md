### 2.3 移动端检测

```typescript
// composables/useDevice.ts
// 设备类型检测：基于视口宽度判断是否为移动端
export const useDevice = () => {
  const isMobile = ref(false);

  onMounted(() => {
    // 检测函数：阈值 768px 与移动端断点一致
    const check = () => {
      isMobile.value = window.innerWidth < 768;
    };
    check();
    window.addEventListener('resize', check);
    onUnmounted(() => window.removeEventListener('resize', check));
  });

  return { isMobile };
};
```

## 三、子项目构建规范

### 3.1 强制 Hash 路由

每个需要嵌入到 iframe 的子项目，必须使用 Hash 路由模式：

```javascript
// 各子项目 router/index.js 需要修改
const router = createRouter({
  history: createWebHashHistory(), // ← 强制 Hash 模式
  routes,
});
```

对不支持 Vue Router 的项目（如纯 HTML Demo），将 HTML/CSS/JS 打包为单个文件，不依赖路由。

### 3.2 构建命令

```bash
# 各子项目目录下执行
cd F:/Project/TourismAnalysis/frontend
VITE_ROUTER_MODE=hash npm run build

# 复制到博客 public/demos/
Copy-Item -Path dist\* -Destination F:\Project\oh-my-blog\frontend\public\demos\tourism-analysis\ -Recurse -Force
```

## 四、样式冲突隔离

### 4.1 iframe 隔离（V1 默认方案）

iframe 内的子项目与博客框架完全隔离——CSS、JS、DOM 都独立。**V1 不额外处理样式冲突**。

### 4.2 sandbox 属性说明

```html
<iframe
  sandbox="allow-scripts allow-same-origin"
  <!-- allow-scripts: 允许 JS 执行（ECharts 图表需要） -->
  <!-- allow-same-origin: 允许同源请求（Demo 数据 fetch 需要） -->
  <!-- 不添加 allow-top-navigation: 防止 iframe 篡改地址栏 -->
  <!-- 不添加 allow-forms: V1 Demo 不需要表单 -->
/>
```

### 4.3 V2 优化方向

当 iframe 的以下问题变得不可接受时，考虑迁移：
1. **暗色主题同步** → PostMessage 通信
2. **加载速度** → Shadow DOM 嵌套（仅简单项目如 TaoBaoUserDrow）
3. **SEO** → 为 iframe 页面添加独立的 `<noscript>` fallback

### 6.3 Markdown 渲染配置

```typescript
// composables/useMarkdown.ts
// Markdown 渲染与目录提取：支持代码高亮、Mermaid 图表
import MarkdownIt from 'markdown-it';
import hljs from 'highlight.js';
import markdownItMermaid from 'markdown-it-mermaid';

export const useMarkdown = () => {
  // 初始化 MarkdownIt：启用 HTML、自动链接、排版优化
  const md = new MarkdownIt({
    html: true,
    linkify: true,
    typographer: true,
    highlight(str: string, lang: string) {
      // 若语言被 highlight.js 支持，则执行语法高亮
      if (lang && hljs.getLanguage(lang)) {
        return `<pre class="hljs"><code>${hljs.highlight(str, { language: lang }).value}</code></pre>`;
      }
      return `<pre class="hljs"><code>${md.utils.escapeHtml(str)}</code></pre>`;
    },
  }).use(markdownItMermaid);

  // 渲染原始 Markdown 为 HTML
  const render = (content: string) => md.render(content);

  // 提取文章目录：基于 H2/H3 标题生成导航树
  const extractTOC = (content: string) => {
    const headings = content.match(/^#{2,3}\s+.+$/gm) || [];
    return headings.map(h => ({
      level: h.startsWith('### ') ? 3 : 2,
      text: h.replace(/^#+\s+/, ''),
      id: h.replace(/^#+\s+/, '').toLowerCase().replace(/\s+/g, '-'),
    }));
  };

  return { render, extractTOC };
};
```
